```sql
SELECT c.CustomerId, c.FirstName, c.LastName, l.LoanId, l.Principal, l.Status
FROM dbo.Customer c
JOIN dbo.Loan l ON l.CustomerId = c.CustomerId AND l.Status = 'active';

CREATE VIEW dbo.vw_Loan_Overdue AS
SELECT l.LoanId, l.CustomerId, p.DueDate, p.AmountDue, p.PaidAmount,
       DATEDIFF(DAY, p.DueDate, ISNULL(p.PaidDate, CAST(SYSUTCDATETIME() AS DATE))) AS DaysPastDue
FROM dbo.Loan l
JOIN dbo.Payment p ON p.LoanId = l.LoanId
WHERE (p.PaidDate IS NULL AND p.DueDate < CAST(SYSUTCDATETIME() AS DATE))
   OR (p.PaidAmount < p.AmountDue AND p.DueDate < CAST(SYSUTCDATETIME() AS DATE));
```
